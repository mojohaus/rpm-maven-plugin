22  import java.io.ByteArrayInputStream;
23  import java.io.ByteArrayOutputStream;
24  import java.io.File;
25  import java.io.IOException;
26  import java.io.InputStream;
27  import java.io.OutputStreamWriter;
28  import java.io.PrintWriter;
29  
30  import org.apache.maven.plugin.logging.Log;
31  import org.codehaus.plexus.util.cli.CommandLineException;
32  import org.codehaus.plexus.util.cli.CommandLineUtils;
33  import org.codehaus.plexus.util.cli.Commandline;
34  import org.codehaus.plexus.util.cli.StreamConsumer;
35  
36  /**
37   * Utility that uses <a href="http://expect.nist.gov/">expect</a> to sign rpms automatically (no user interaction).
38   *
39   * @author Brett Okken, Cerner Corporation
40   * @since 2.0-beta-4
41   */
42  final class RPMSigner
43  {
44      /**
45       * GPG path.
46       */
47      private final File gpgPath;
48  
49      /**
50       * GPG name as defined in the rpm database.
51       */
52      private final String gpgName;
53  
54      /**
55       * The passphrase for the gpg key.
56       */
57      private final char[] passphrase;
58  
59      /**
60       * {@code Log} to log to.
61       */
62      private final Log log;
63  
64      /**
65       * Constructor takes all necessary attributes to sign an rpm.
66       *
67       * @param gpgPath Directory containing key data.
68       * @param gpgName The name of the gpg key in the rpm database.
69       * @param passphrase The passphrase for the gpg key.
70       * @param log Used for logging information in the signing process.
71       */
72      public RPMSigner( File gpgPath, String gpgName, char[] passphrase, Log log )
73      {
74          this.gpgPath = gpgPath;
75          this.gpgName = gpgName;
76          this.passphrase = passphrase;
77          this.log = log;
78      }
79  
80      /**
81       * Signs the rpm using the gpgName and passphrase given. All output from the signing will be written to <i>log</i>.
82       *
83       * @param rpm RPM file to sign. Must exist and be readable.
84       * @throws IOException
85       */
86      public void sign( final File rpm )
87          throws IOException
88      {
89          if ( !rpm.exists() || !rpm.canRead() )
90          {
91              throw new IllegalStateException( rpm.getAbsolutePath() + " is not a valid rpm file or cannot be read" );
92          }
93  
94          // use option to provide "script" via stdin
95          final Commandline cl = new Commandline();
96          cl.setExecutable( "expect" );
97          cl.setWorkingDirectory( rpm.getParentFile() );
98          cl.createArg().setValue( "-c" );
99          //work around to allow expect execution time to read in the input during heavy system usage
100         cl.createArg().setValue( "sleep 1" ); //MRPM-176 and PLXUTILS-174
101         cl.createArg().setValue( "-" );
102 
103         final StreamConsumer stdout = new LogStreamConsumer( LogStreamConsumer.INFO, log );
104         final StreamConsumer stderr = new LogStreamConsumer( LogStreamConsumer.WARN, log );
105 
106         try
107         {
108             if ( log.isDebugEnabled() )
109             {
110                 log.debug( "About to execute \'" + cl.toString() + "\'" );
111             }
112 
113             final InputStream is = new ByteArrayInputStream( writeExpectScriptFile( rpm ) );
114 
115             int result = CommandLineUtils.executeCommandLine( cl, is, stdout, stderr );
116             if ( result != 0 )
117             {
118                 throw new IllegalStateException( "RPM sign execution returned: \'" + result + "\' executing \'"
119                     + cl.toString() + "\'" );
120             }
121         }
122         catch ( CommandLineException e )
123         {
124             final IllegalStateException ise = new IllegalStateException( "Unable to sign the RPM" );
125             ise.initCause( e );
126             throw ise;
127         }
128     }
129 
130     /**
131      * Writes the expect "script".
132      *
133      * @param rpm The rpm to sign.
134      * @return The expect script as a {@code byte[]}.
135      * @throws IOException
136      */
137     private byte[] writeExpectScriptFile( final File rpm )
138         throws IOException
139     {
140         final ByteArrayOutputStream baos = new ByteArrayOutputStream( 512 );
141 
142         final PrintWriter writer = new PrintWriter( new OutputStreamWriter( baos ) );
143         try
144         {
145             writer.println( "set timeout -1" );
146             writer.print( "spawn rpm --define \"_gpg_name " );
147             writer.print( gpgName );
148             writer.print( "\"" );
149             if ( gpgPath != null )
150             {
151                 writer.print( " --define \"_gpg_path " );
152                 writer.print( gpgPath + "\"" );
153             }
154             writer.print( " --addsign " );
155             writer.println( rpm.getName() );
156             writer.println( "expect -exact \"Enter pass phrase: \"" );
157             writer.print( "send -- \"" );
158             writer.print( new String( this.passphrase ) );
159             writer.println( "\r\"" );
160             writer.println( "expect {" );
161             writer.println( " \"Pass phrase is good.\" {" );
162             writer.println( "      expect eof" );
163             writer.println( "      exit 0" );
164             writer.println( "  }" );
165             writer.println( " \"signing failed\" {" );
166             writer.println( "      expect eof" );
167             writer.println( "      exit 1" );
168             writer.println( "  }" );
169             writer.println( "}" );
170             writer.println();
171             writer.flush();
172         }
173         finally
174         {
175             writer.close();
176         }
177 
178         return baos.toByteArray();
179     }
180 }
